# Functions to create boolean vectors for segmenting 
# the tweet data based on tweet content:

def photo_embedded(media_list):
    try:
        for m in media_list:
            if m['type'] == 'photo':
                return True
    except TypeError: 
        return False
    
def get_boolvecs_phu(tweets):
    """
    Segment the data into disjoint sets depending on their features.

    'tweets' is a Pandas DataFrame of tweets loaded from a json file.
    """

    contains_photo = [photo_embedded(ml) for ml in tweets['media']]
    contains_hashtags = [h != [] for h in tweets['hashtags']]
    contains_url = [any(u) for u in tweets['urls']]
    all_boolvec = zip(contains_photo, contains_hashtags, contains_url)

    contains_photo_only    = [p and all((not h,not u)) for p,h,u in all_boolvec]
    contains_hashtags_only = [h and all((not p,not u)) for p,h,u in all_boolvec]
    contains_url_only      = [u and all((not p,not h)) for p,h,u in all_boolvec]

    contains_hashtags_url_only   = [all((h,u)) and not p for p,h,u in all_boolvec]
    contains_photo_url_only      = [all((p,u)) and not h for p,h,u in all_boolvec]
    contains_photo_hashtags_only = [all((p,h)) and not u for p,h,u in all_boolvec]

    contains_all =  [all((p,h,u)) for p,h,u in all_boolvec]
    contains_none = [any(phu) == False for phu in zip(contains_photo, contains_hashtags, contains_url)]
    
    return contains_photo_only, contains_hashtags_only, contains_url_only, \
        contains_hashtags_url_only, contains_photo_url_only, contains_photo_hashtags_only, \
        contains_all, contains_none

tweets_sq_morethan5_orig = tweets_sq_morethan5[tweets_sq_morethan5['summedDC'] != 0]

# group the data into 10 disjoint quantiles based on summedDC:
quantile_lowerlimits = [i/10.0 for i in range(10)]
quantilegroups_boolvecs = []

for ql in quantile_lowerlimits:
    quantilelim_lower = tweets_sq_morethan5_orig.quantile(ql)['summedDC']
    quantilelim_upper = tweets_sq_morethan5_orig.quantile(ql+0.1)['summedDC']

    if ql != max(quantile_lowerlimits):
        quantilegroups_boolvecs.append([sdc >= quantilelim_lower and sdc < quantilelim_upper \
                                  for sdc in tweets_sq_morethan5_orig['summedDC']])
    else:
        quantilegroups_boolvecs.append([sdc >= quantilelim_lower and sdc <= quantilelim_upper \
                                  for sdc in tweets_sq_morethan5_orig['summedDC']])
        
# A list of 10 DataFrames containing the data broken into 10 quantiles
# on the summedDC calculation:
quantiles = [tweets_sq_morethan5_orig[bv] for bv in quantilegroups_boolvecs]
